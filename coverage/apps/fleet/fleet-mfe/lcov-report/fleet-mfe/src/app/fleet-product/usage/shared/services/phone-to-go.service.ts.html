import { Injectable } from '@angular/core';
import { Observable, Subject } from 'rxjs';
import { OcapiCallConfig, OcapiService } from '@telenet/ng-lib-ocapi';
import { ProductConstants } from '../constants/product.constant';
import { PhoneToGoResponseMapper } from '../mappers/phone-to-go-response-mapper';
import { PhoneToGoConfigurationModel } from '../interfaces/phone-to-go.interface';
 
@Injectable({
  providedIn: 'root',
})
export class PhoneToGoService {
  private readonly ENDPOINT = ProductConstants.PRODUCT_RESOURCE_SERVICE_PREFIX + 'v1/cloud-telephony/phone2go';
  private readonly subject = new Subject<unknown>();
 
  constructor(
    private readonly ocapiService: OcapiService,
    private readonly phoneToGoResponseMapper: PhoneToGoResponseMapper
  ) {}
 
  getPhoneToGoConfiguration(messageGroup: string): Observable<PhoneToGoConfigurationModel> {
    const ocapiCallConfig = new OcapiCallConfig(messageGroup, this.ENDPOINT, this.phoneToGoResponseMapper);
    return this.ocapiService.doGet(ocapiCallConfig);
  }
}
