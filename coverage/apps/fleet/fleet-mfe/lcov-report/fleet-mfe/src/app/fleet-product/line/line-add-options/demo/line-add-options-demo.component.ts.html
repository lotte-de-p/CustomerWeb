import { Component, inject, OnInit } from '@angular/core';
 
import { FleetManagerLineAddOptionsComponent } from '../fleet-manager-line-add-options.component';
import { FleetManagerLineOverlayComponent } from '../../line-overlay/fleet-manager-line-overlay.component';
import { LineAddOptionsConstants } from '../../shared/constants/line-add-options.constants';
import { ProductsService } from '../../shared/services/product/products.service';
import { ProductSelectedLineService } from '../../shared/services/product/product-selected-line-service';
 
@Component({
  selector: 'tg-line-add-options-demo',
  standalone: true,
  styles: [],
  imports: [FleetManagerLineAddOptionsComponent, FleetManagerLineOverlayComponent],
  template: `<tg-fleet-manager-line-overlay>
    <tg-fleet-manager-line-add-options [msisdn]="msisdn"></tg-fleet-manager-line-add-options>
  </tg-fleet-manager-line-overlay>`,
})
export class LineAddOptionsDemoComponent implements OnInit {
  readonly msisdn: string = '0485012038';
  readonly #productSelectedLineService: ProductSelectedLineService = inject(ProductSelectedLineService);
  readonly #productService = inject(ProductsService);
 
  ngOnInit(): void {
    this.#productService
      .getProductByIdentifier(LineAddOptionsConstants.MESSAGE_GROUP, this.msisdn)
      .subscribe((response) => {
        this.#productSelectedLineService.setSelectedLines([response]);
      });
  }
}
