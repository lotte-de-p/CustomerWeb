import { Injectable } from '@angular/core';
import isUndefined from 'lodash-es/isUndefined';
import each from 'lodash-es/each';
import { PhoneNumberTransformService } from '@telenet/ng-lib-layout';
import isEmpty from 'lodash-es/isEmpty';
import flatten from 'lodash-es/flatten';
import { CustomerProductHolding } from '../models/customer-product-holding.model';
import { ProductConstants } from '../constants/product.constant';
import { EasySwitchDetail } from '../models/easy-switch-detail.model';
import { ProductsStore } from '../models/products-store.model';
import { PhoneToGoConfigurationModel } from '../interfaces/phone-to-go.interface';
import { ProductSubscriptionsInterface } from '../interfaces/product-subscriptions.interface';
import { Builder } from '@telenet/ng-lib-shared';
import { TIPContractInterface } from '../interfaces/tip-contract.interface';
 
@Injectable({
  providedIn: 'root',
})
export class ProductsStoreMapper {
  constructor(private readonly phoneNumberTransformService: PhoneNumberTransformService) {}
 
  private readonly MOBILE_CATEGORIES = ['prepaid-mobile', 'post-paid-mobile', 'base-pro'];
  private readonly MOBILE = 'mobile';
  private readonly TELEPHONE = 'telephone';
  private readonly DTV = 'dtv';
  private readonly INTERNET = 'internet';
  private readonly PHONE2GO = 'phone2go';
 
  public toProductStore(
    loggedInMsisdn: string,
    productHoldings: CustomerProductHolding[],
    pendingProducts: ProductSubscriptionsInterface[],
    activeTIPContracts: TIPContractInterface[],
    easySwitchDetail: EasySwitchDetail[]
  ): ProductsStore {
    const productIDs = this.toProductEPCIds(productHoldings);
    return Builder(ProductsStore)
      .hasOnlyPrepaidProducts(
        this.allProductsHasCategory(productHoldings, ProductConstants.OMAPI_CATEGORY_PRE_PAID_MOBILE)
      )
      .hasOnlyPostpaidProducts(
        this.allProductsHasCategory(productHoldings, ProductConstants.OMAPI_CATEGORY_POST_PAID_MOBILE)
      )
      .loggedInProductCategory(this.getLoggedInUserProductCategory(productHoldings, loggedInMsisdn))
      .productID(productIDs)
      .hasBundleMobileProduct(this.hasBundleMobileProduct(productHoldings))
      .hasStandaloneMobileProduct(this.hasStandaloneMobileProduct(productHoldings))
      .hasFixedTelephone(this.hasProductType(productHoldings, this.TELEPHONE))
      .hasDtvProduct(this.hasProductType(productHoldings, this.DTV))
      .hasInternetProduct(this.hasProductType(productHoldings, this.INTERNET))
      .hasPhone2Go(this.hasProductType(productHoldings, this.PHONE2GO))
      .hasBundleProduct(this.hasBundle(productHoldings))
      .hasMobileLinesNotActivated(this.hasMobileLinesNotActivated(productHoldings))
      .hasInternetLinesNotActivated(this.hasInternetLinesNotActivated(productHoldings))
      .hasEasySwitchId(this.hasEasySwitchId(easySwitchDetail))
      .hasPendingOrders(pendingProducts.length > 0)
      .inactiveIncludedOptions(this.inactiveIncludedOptions(productHoldings))
      .inactiveStreamingOptions(this.inactiveStreamingOptions(productHoldings))
      .activeIncludedOptions(this.activeIncludedOptions(productHoldings))
      .hasEsimAvailable(this.hasEsimAvailable(productHoldings))
      .recentlyActivatedProductAge(this.calculateRecentlyActivatedProductAge(productHoldings))
      .productHoldings(productHoldings)
      .pendingProducts(pendingProducts)
      .hasOnlyActivationInProgressLines(this.hasOnlyActivationInProgressLines(productHoldings))
      .hasOnlyDisconnectedLines(this.hasOnlyDisconnectedLines(productHoldings))
      .hasNotActiveAndPendingActivationMobileLines(this.hasNotActiveAndPendingActivationMobileLines(productHoldings))
      .hasNotActiveAndPendingActivationInternetLines(
        this.hasNotActiveAndPendingActivationInternetLines(productHoldings)
      )
      .hasActiveTIPContracts(activeTIPContracts.length > 0)
      .activeTipContracts(activeTIPContracts)
      .productSubGroups(this.getProductSubGroups(productHoldings))
      .isSuccess((productIDs && productIDs.length > 0) || activeTIPContracts.length > 0)
      .build();
  }
 
  public toProductStoreWithPhone2Go(
    productsStore: ProductsStore,
    phoneToGoConfiguration: PhoneToGoConfigurationModel
  ): ProductsStore {
    productsStore.isWebexEmailLinked = phoneToGoConfiguration.users.some((user) => user.emailAddress.length > 0);
    productsStore.phoneToGoProducts = phoneToGoConfiguration;
    return productsStore;
  }
 
  private calculateRecentlyActivatedProductAge(productHoldings: CustomerProductHolding[]): number {
    let recentlyActivatedProductAge = 9999;
    each(productHoldings, (product) => {
      if (product.activationDate) {
        const calculatedAge = this.getActivatedProductAge(product.activationDate);
        if (calculatedAge < recentlyActivatedProductAge) {
          recentlyActivatedProductAge = calculatedAge;
        }
      }
    });
    return recentlyActivatedProductAge;
  }
 
  private hasEsimAvailable(productHoldings: CustomerProductHolding[]) {
    return productHoldings.some((p) => p.productInfo.getEsimCharacteristicValue());
  }
 
  private inactiveIncludedOptions(productHoldings: CustomerProductHolding[]) {
    return flatten(productHoldings.map((productHolding) => productHolding.options))
      .map((option) => this.isIncludedOptionAndNotActive(option))
      .filter(this.notNull);
  }
 
  private activeIncludedOptions(productHoldings: CustomerProductHolding[]) {
    return flatten(productHoldings.map((productHolding) => productHolding.options))
      .map((option) => this.isIncludedOptionAndActive(option))
      .filter(this.notNull);
  }
 
  private isIncludedOptionAndNotActive(option) {
    return option.isIncludedOption && option.status === 'NOT_ACTIVE' ? option.productInfo.externalProductCode : null;
  }
 
  private isIncludedOptionAndActive(option) {
    return option.isIncludedOption && option.status === 'ACTIVE' ? option.productInfo.externalProductCode : null;
  }
 
  private inactiveStreamingOptions(productHoldings: CustomerProductHolding[]) {
    return flatten(productHoldings.map((productHolding) => productHolding.options))
      .map((option) => this.isOptionWithNotActiveStatus(option))
      .filter(this.notNull);
  }
 
  private isOptionWithNotActiveStatus(option) {
    if (
      option !== undefined &&
      option.status === ProductConstants.PRODUCT_STATUS_ACTIVATION_PASSED_PONR &&
      option.productInfo?.productId?.startsWith(ProductConstants.STREAMING_PRODUCTS_PREFIX)
    ) {
      if (option.chars?.specUrl?.includes(ProductConstants.STREAMING_PRODUCTS_PREFIX)) {
        return option.chars.specUrl
          .split('/')
          .find((param) => param.includes(ProductConstants.STREAMING_PRODUCTS_PREFIX));
      } else {
        return option.productInfo.externalProductCode;
      }
    }
    return null;
  }
 
  private hasProductType(productHoldings: CustomerProductHolding[], productType: string) {
    return productHoldings.filter(this.checkAllowedStatuses).some((p) => p.hasProductType(productType));
  }
 
  private checkAllowedStatuses(product: CustomerProductHolding): boolean {
    return ['ACTIVE', 'SUSPENDED', 'ACTIVATION_IN_PROGRESS', 'ACTIVATION_PASSED_PONR'].includes(product.status);
  }
 
  private hasMobileLinesNotActivated(productHoldings: CustomerProductHolding[]) {
    return productHoldings.some(
      (p) =>
        p.hasProductTypeAndStatus(this.MOBILE, [ProductConstants.PRODUCT_STATUS_PENDING]) ||
        p.options.some((o) => o.productType === this.MOBILE && o.status === ProductConstants.PRODUCT_STATUS_PENDING)
    );
  }
 
  private hasInternetLinesNotActivated(productHoldings: CustomerProductHolding[]) {
    return productHoldings.some(
      (p) =>
        p.hasProductTypeAndStatus(this.INTERNET, [ProductConstants.PRODUCT_STATUS_PENDING]) ||
        p.options.some((o) => o.productType === this.INTERNET && o.status === ProductConstants.PRODUCT_STATUS_PENDING)
    );
  }
 
  private hasEasySwitchId(easySwitchDetail: EasySwitchDetail[]): boolean {
    return !isEmpty(easySwitchDetail) && easySwitchDetail.filter((x) => x.id).length > 0;
  }
 
  private hasBundle(productHoldings: CustomerProductHolding[]) {
    return productHoldings.some((p) => p.isBundleProduct());
  }
 
  private allProductsHasCategory(productHoldings: CustomerProductHolding[], productCategory: string): boolean {
    return productHoldings.every((p) => !isUndefined(p.productInfo) && p.productInfo.hasCategory(productCategory));
  }
 
  private getLoggedInUserProductCategory(productHoldings: CustomerProductHolding[], msisdn: string = ''): string {
    const normalizedMsisdn = this.normalizedMsisdn(msisdn);
    const loggedInProductHolding = productHoldings.find(
      (p) => p.identifier.replace(/\s/g, '') === normalizedMsisdn.replace(/\s/g, '')
    );
    if (loggedInProductHolding && loggedInProductHolding.productInfo) {
      return this.MOBILE_CATEGORIES.find((c) => loggedInProductHolding.productInfo.hasCategory(c));
    }
  }
 
  private normalizedMsisdn(msisdn: string): string {
    return this.phoneNumberTransformService.transform(msisdn);
  }
 
  toProductEPCIds(productHoldings: CustomerProductHolding[]): string[] {
    const productEpcs = productHoldings.map((p) => p.productInfo.externalProductCode).filter(this.notNull);
    const optionEpcs = flatten(productHoldings.map((p) => p.options))
      .map((option) => option.productInfo && option.productInfo.externalProductCode)
      .filter(this.notNull);
    const childrenEpcs = flatten(productHoldings.map((p) => p.children))
      .map((children) => children && children.productInfo && children.productInfo.externalProductCode)
      .filter(this.notNull);
    const childOptionIds = flatten(this.allChildOptions(productHoldings))
      .map((option) => option.productInfo && option.productInfo.externalProductCode)
      .filter(this.notNull);
    const experienceIds = flatten(productHoldings.map((p) => p.children))
      .map((children) => children && children.experience && children.experience.externalProductCode)
      .filter(this.notNull);
 
    return productEpcs.concat(optionEpcs).concat(childrenEpcs).concat(childOptionIds).concat(experienceIds).sort();
  }
 
  private allChildOptions(productHoldings: CustomerProductHolding[]) {
    return flatten(productHoldings.map((p) => p.children).filter(this.notNull)).map((child) => child.options);
  }
 
  private getProductSubGroups(productHoldings: CustomerProductHolding[]): string[] {
    return productHoldings.map((p) => p.productInfo && p.productInfo.getProductSubGroup()).filter(this.notNull);
  }
 
  private notNull<T>(value: T | null | undefined): value is T {
    return value !== null && value !== undefined;
  }
 
  private getActivatedProductAge(activationDate: string): number {
    const productActivationDate = new Date(activationDate);
    const currentDate = new Date();
 
    return Math.floor((currentDate.getTime() - productActivationDate.getTime()) / 1000 / 60 / 60 / 24);
  }
 
  private hasNotActiveAndPendingActivationInternetLines(productHoldings: CustomerProductHolding[]): boolean {
    if (!this.hasActiveInternetLines(productHoldings)) {
      return this.hasInternetLinesNotActivated(productHoldings);
    }
  }
 
  private hasNotActiveAndPendingActivationMobileLines(productHoldings: CustomerProductHolding[]): boolean {
    if (!this.hasActiveMobileLines(productHoldings)) {
      return this.hasMobileLinesNotActivated(productHoldings);
    }
  }
 
  private hasActiveMobileLines(productHoldings: CustomerProductHolding[]) {
    return productHoldings.some(
      (p) =>
        p.hasProductTypeAndStatus(this.MOBILE, [ProductConstants.PRODUCT_STATUS_ACTIVE]) ||
        p.options.some((o) => o.productType === this.MOBILE && o.status === ProductConstants.PRODUCT_STATUS_ACTIVE)
    );
  }
 
  private hasActiveInternetLines(productHoldings: CustomerProductHolding[]) {
    return productHoldings.some(
      (p) =>
        p.hasProductTypeAndStatus(this.INTERNET, [ProductConstants.PRODUCT_STATUS_ACTIVE]) ||
        p.options.some((o) => o.productType === this.MOBILE && o.status === ProductConstants.PRODUCT_STATUS_ACTIVE)
    );
  }
 
  private hasActiveLines(productHoldings: CustomerProductHolding[]): boolean {
    return productHoldings.some((p) => p.hasProductStatus(ProductConstants.PRODUCT_STATUS_ACTIVE));
  }
 
  private hasDisconnectedLines(productHoldings: CustomerProductHolding[]): boolean {
    return productHoldings.some(
      (p) =>
        p.hasProductStatus(ProductConstants.PRODUCT_STATUS_DISCONNECTED) ||
        p.hasProductStatus(ProductConstants.PRODUCT_STATUS_DISCONNECTION_IN_PROGRESS)
    );
  }
 
  private hasActivationInProgressLines(productHoldings: CustomerProductHolding[]): boolean {
    return productHoldings.some(
      (p) =>
        p.hasProductStatus(ProductConstants.PRODUCT_STATUS_PENDING) ||
        p.hasProductStatus(ProductConstants.PRODUCT_STATUS_ACTIVATION_PASSED_PONR)
    );
  }
 
  private hasOnlyDisconnectedLines(productHoldings: CustomerProductHolding[]): boolean {
    if (!this.hasActiveLines(productHoldings) && !this.hasActivationInProgressLines(productHoldings)) {
      return this.hasDisconnectedLines(productHoldings);
    }
 
    return false;
  }
 
  private hasOnlyActivationInProgressLines(productHoldings: CustomerProductHolding[]): boolean {
    if (!this.hasActiveLines(productHoldings)) {
      return this.hasActivationInProgressLines(productHoldings);
    }
 
    return false;
  }
 
  private hasBundleMobileProduct(productHoldings: CustomerProductHolding[]) {
    return productHoldings
      .filter((product) => product.isBundleProduct())
      .some((product) => this.hasProducts(this.MOBILE, product));
  }
 
  private hasStandaloneMobileProduct(productHoldings: CustomerProductHolding[]) {
    return productHoldings
      .filter((product) => !product.productType || !product.isBundleProduct())
      .some((product) => !product.productType || this.hasProducts(this.MOBILE, product));
  }
 
  private hasProducts(productType: string, productHolding: CustomerProductHolding) {
    return productHolding.hasProductTypeAndStatus(productType, [
      ProductConstants.PRODUCT_STATUS_ACTIVE,
      ProductConstants.PRODUCT_STATUS_SUSPENDED,
      ProductConstants.PRODUCT_STATUS_PENDING,
      ProductConstants.PRODUCT_STATUS_ACTIVATION_PASSED_PONR,
    ]);
  }
}
